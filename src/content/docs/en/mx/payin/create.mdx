import ReqAddr from "@components/ReqAddr.astro";
import Table from "@components/Table.astro";

### 请求地址

<ReqAddr method="POST" url="/api/pay/payment/create/v1" />

### 请求参数

<Table
  thead={["字段", "类型", "必需", "长度", "描述"]}
  tbody={[
    ["merchantOrderNo", "String", "yes", "32" ,"商户订单号"],
    ["paymentType", "int", "yes", "", "支付方式: 1-还款码 2-收银台"],
    ["amount", "String", "yes", "20" ,"代收金额(比索)"],
    ["callbackUrl", "String", "no", "200", "代付回调地址，若不传, 则以商户配置为准"],
    ["sign", "String", "yes", "","签名"],
  ]}
/>

```json title=请求示例
{
  "merchantOrderNo": "201806251011",
  "paymentType": 1,
  "amount": "100",
  "callbackUrl": "https://merchant.com/api/payment/callback",
  "sign": "YOUR_SIGN"
}
```

### 返回参数

<Table
  thead={["字段", "类型", "必需", "描述"]}
  tbody={[
    ["merchantOrderNo", "String", "yes", "商户订单号(Max leng 32)"],

```json title=返回示例
  "tradeNo": "TF201806251011",
  "method": "VA",

  "url": "",
  "sign": "TEEMO_SIGN"
}
```
